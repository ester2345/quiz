document.getElementById('quiz-form').addEventListener('submit', function(event) { event.preventDefault(); // Evita o envio do formulário const correctAnswers = { q1: 'Brasília', q2: 'Machado de Assis', q3: 'Júpiter' }; const resultDiv = document.getElementById('result'); resultDiv.innerHTML = ''; // Limpa resultados anteriores for (const [question, correctAnswer] of Object.entries(correctAnswers)) { const selectedAnswer = document.querySelector(`input[name="${question}"]:checked`); if (selectedAnswer) { const label = selectedAnswer.parentElement; if (selectedAnswer.value === correctAnswer) { label.classList.add('correct'); resultDiv.innerHTML += `

${question}: Resposta correta!

`; } else { label.classList.add('incorrect'); resultDiv.innerHTML += `

${question}: Resposta errada! A correta é ${correctAnswer}.

`; } } else { resultDiv.innerHTML += `

${question}: Nenhuma resposta selecionada.

`; } } });
